### Python CDP 範例
```python
import asyncio
import json
import websockets

async def click_button():
    uri = "ws://127.0.0.1:9222/devtools/page/{targetId}"
    async with websockets.connect(uri) as ws:
        script = """
        (() => {
            const btns = Array.from(document.querySelectorAll('tp-yt-paper-button'));
            for (const btn of btns) {
                if (btn.textContent.includes('顯示完整內容')) {
                    btn.click();
                    return 'clicked';
                }
            }
            return 'not found';
        })()
        """
        await ws.send(json.dumps({
            "id": 1,
            "method": "Runtime.evaluate",
            "params": {"expression": script, "returnByValue": True}
        }))
        response = await ws.recv()
        print(response)
```
